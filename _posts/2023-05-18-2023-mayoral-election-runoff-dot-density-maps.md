```js
// turf via CDN — used by the dot generator below.
const turf = await import("https://esm.sh/@turf/turf@7");
```

```js
// Round a fractional dot count, keeping the expected value (so e.g. 2.3 dots
// becomes 2 or 3 at random rather than always truncating).
const randround = (val) => Math.floor(val + Math.random());
```

```js
// Scatter n uniform-random points inside a polygon feature by rejection
// sampling its bounding box. This replaces the original notebook's imported
// dot-density generator (Ben Schmidt's earcut/Arrow/WebGL trifeather stack) —
// a few lines of turf do the same job with no triangulation or GPU.
function randomPointsInPolygon(n, feature) {
  const [minX, minY, maxX, maxY] = turf.bbox(feature);
  const points = [];
  let guard = 0;
  while (points.length < n && guard < n * 200 + 100) {
    guard++;
    const p = [
      minX + Math.random() * (maxX - minX),
      minY + Math.random() * (maxY - minY),
    ];
    if (turf.booleanPointInPolygon(p, feature)) points.push(p);
  }
  return points;
}
```
